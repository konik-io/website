019package io.konik.zugferd.unqualified;
020
021import java.io.Serializable;
022
023import javax.xml.bind.annotation.XmlAttribute;
024import javax.xml.bind.annotation.XmlSchemaType;
025import javax.xml.bind.annotation.XmlType;
026import javax.xml.bind.annotation.XmlValue;
027import javax.xml.bind.annotation.adapters.CollapsedStringAdapter;
028import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
029
030import io.konik.validator.annotation.NotBlank;
031
032/**
033 * = The classification code
034 */
035@XmlType(name = "CodeType", propOrder = { "value" })
036public class Code implements Serializable {
037
038   @XmlValue
039   @XmlJavaTypeAdapter(CollapsedStringAdapter.class)
040   @XmlSchemaType(name = "token")
041   private String value;
042
043   @XmlAttribute(name = "listID")
044   @XmlJavaTypeAdapter(CollapsedStringAdapter.class)
045   @XmlSchemaType(name = "token")
046   private String listId;
047
048   @XmlAttribute(name = "listVersionID")
049   @XmlJavaTypeAdapter(CollapsedStringAdapter.class)
050   @XmlSchemaType(name = "token")
051   private String listVersionId;
052
053   /**
054    * Gets the value.
055    *
056    * @return the value
057    */
058   @NotBlank
059   public String getValue() {
060      return value;
061   }
062
063   /**
064    * Sets the value.
065    *
066    * @param value the value
067    */
068   public void setValue(String value) {
069      this.value = value;
070   }
071
072   /**
073    * Gets the list id.
074    *
075    * @return the list id
076    */
077   @NotBlank
078   public String getListID() {
079      return listId;
080   }
081
082   /**
083    * Sets the list id.
084    *
085    * @param value the list id
086    */
087   public void setListID(String value) {
088      this.listId = value;
089   }
090
091   /**
092    * Gets the list version id.
093    *
094    * @return the list version id
095    */
096   public String getListVersionID() {
097      return listVersionId;
098   }
099
100   /**
101    * Sets the list version id.
102    *
103    * @param value the list version id
104    */
105   public void setListVersionID(String value) {
106      this.listVersionId = value;
107   }
108
109}
