019package io.konik.jaxb.bindable.entity;
020
021import static io.konik.util.Strings.isNotEmpty;
022import static io.konik.util.Strings.isNullOrEmpty;
023
024import java.util.regex.Pattern;
025
026import javax.xml.bind.annotation.adapters.XmlAdapter;
027
028/**
029 * = The Universal Communication Adapter
030 */
031public class UniversalCommunicationAdapter extends XmlAdapter<UniversalCommunication, String> {
032
033   Pattern emailRegEx = Pattern.compile(".+@.+\\..+");
034
035   @Override
036   public UniversalCommunication marshal(String unversalCommunication) throws Exception {
037      if (isNullOrEmpty(unversalCommunication)) { return null; }
038
039      UniversalCommunication communication = new UniversalCommunication();
040      if (emailRegEx.matcher(unversalCommunication).matches()) {
041         communication.setMail(unversalCommunication);
042      } else {
043         communication.setCompleteNumber(unversalCommunication);
044      }
045      return communication;
046   }
047
048   @Override
049   public String unmarshal(UniversalCommunication communication) throws Exception {
050      if (isNotEmpty(communication.getCompleteNumber())) { return communication.getCompleteNumber(); }
051      return communication.getMail();
052   }
053}
