018package io.konik.jaxb.bindable.unqualified;
019
020import io.konik.zugferd.unqualified.ZfDate;
021import io.konik.zugferd.unqualified.ZfDateFactory;
022
023import javax.xml.bind.annotation.adapters.XmlAdapter;
024
025/**
026 * = The Event to ZUGFeRD Date Adapter.
027 */
028public class DateTimeAdapter extends XmlAdapter<DateTime, ZfDate> {
029
030   @Override
031   public ZfDate unmarshal(DateTime dateTime) throws Exception {
032      if (dateTime == null) { return null; }
033      ZfDate zfDate = ZfDateFactory.create(dateTime.dateTimeString.format);
034      zfDate.setDate(dateTime.dateTimeString.value);
035      return zfDate;
036   }
037
038   @Override
039   public DateTime marshal(ZfDate date) throws Exception {
040      if (date == null) { return null; }
041      return new DateTime(date.toString(), date.getFormatCode());
042   }
043
044}
